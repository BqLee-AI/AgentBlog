## 三、接口测试：用真实 app（`tests/helpers/app.ts`）

> 💡 Hono app 可直接当函数调用（`app.request(path, options)`），无需起真实端口。

```ts
import { app } from '@/app'
import { db } from '@/db/client'
import { users } from '@/db/schema'
import { hashPassword } from '@/lib/hash'
import { eq } from 'drizzle-orm'

let testToken: string

export async function setupTestUser(role: 'user' | 'admin' | 'super_admin' = 'user') {
  const username = `test_${Date.now()}`
  const [user] = await db.insert(users).values({
    username, passwordHash: await hashPassword('test123456'),
    role, credits: 1000,
  }).returning()

  const res = await app.request('/api/auth/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ username, password: 'test123456' }),
  })
  const json = await res.json()
  testToken = json.data.token
  return { user, token: testToken }
}

export function authHeaders(token = testToken) {
  return { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' }
}
```


### 4.5 slug 不可变

```ts
it('更新文章不能改 slug', async () => {
  const created = await postService.create({ title: '原标题', content: 'x', status: 'published', tagIds: [] }, 1, 'user')
```

### 5.2 编程式联调脚本

```ts
// tests/mcp.e2e.test.ts
import { Client } from '@modelcontextprotocol/sdk/client/index.js'
import { StreamableHTTPClientTransport } from '@modelcontextprotocol/sdk/client/streamableHttp.js'

it('MCP 端到端', async () => {
  const transport = new StreamableHTTPClientTransport(
    new URL('http://localhost:3000/mcp'),
    { requestInit: { headers: { 'X-API-Key': process.env.TEST_API_KEY! } } },
  )
  const client = new Client({ name: 'test', version: '1.0' })
  await client.connect(transport)

  const tools = await client.listTools()
  expect(tools.tools.map((t) => t.name)).toEqual([
    'list_posts', 'get_post', 'create_post', 'update_post', 'delete_post',
  ])

  const result = await client.callTool({ name: 'list_posts', arguments: { limit: 5 } })
  expect(result.content[0].type).toBe('text')
})
```
