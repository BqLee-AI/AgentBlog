# 06 · 文章模块

> 本篇实现需求 §4.2「文章管理」：CRUD、slug 永久化、作者归属、标签多对多、草稿/发布。
> 对应任务 7「文章 CRUD 接口」+ 任务 8「标签与文章-标签关联」。
>
> **强约束**：文章一经发布 slug 永久不变（需求 §1.2/§5）；作者类型 user/agent 区分。

---

## 一、接口列表

| 方法 | 路径 | 说明 | 鉴权 |
|------|------|------|------|
| `POST` | `/api/posts` | 创建文章（含草稿/发布） | 登录 |
| `GET` | `/api/posts` | 列表（支持标签/状态/作者筛选、分页） | 公开（仅 published） |
| `GET` | `/api/posts/:slug` | 详情（按 slug） | 公开 |
| `GET` | `/api/posts/by-id/:id` | 详情（按 id，后台编辑用） | 登录 |
| `PATCH` | `/api/posts/:id` | 更新（slug 不可改） | 自己/admin+ |
| `DELETE` | `/api/posts/:id` | 删除 | 自己/admin+ |

> 💡 阅读页用 `:slug`（稳定引用），后台编辑用 `:id`（含草稿）。MCP `get_post` 同时支持 slug 和 id。

```ts
  status: z.enum(['draft', 'published']).optional(),
  tagIds: z.array(z.number().int().positive()).optional(),
})

export const listPostsQuerySchema = z.object({
  page: z.coerce.number().int().min(1).default(1),
  pageSize: z.coerce.number().int().min(1).max(50).default(10),

  status: z.enum(['draft', 'published']).optional(),
  authorType: z.enum(['user', 'agent']).optional(),
  authorId: z.coerce.number().int().optional(),
```

## 三、slug 生成（`src/lib/slug.ts`）

> 📌 slug 用于 Agent 与人长期引用，必须稳定。

```ts
import { randomBytes } from 'node:crypto'

/**
 * 从标题生成 slug。中文标题用时间戳+短随机后缀保证唯一，避免拼音转换依赖。
 * 形如：hello-world 或 1720412345-a3f9k2
 */
export function generateSlug(title: string): string {
  const base = title
    .toLowerCase()
    .trim()
    .replace(/[^\w\u4e00-\u9fa5\s-]/g, '')   // 保留字母数字中文空格连字符
    .replace(/[\s_]+/g, '-')
    .replace(/-+/g, '-')
    .replace(/^-|-$/g, '')

  // 纯英文/数字直接用；否则用短 ID
  const isAscii = /^[\w-]+$/.test(base)
  const stem = isAscii && base ? base : Date.now().toString(36)
  const suffix = randomBytes(3).toString('hex')   // 防碰撞
  return `${stem}-${suffix}`.slice(0, 80)
}
```

> 💡 slug 生成策略可换：拼音库（`pinyin`）/ nanoid / Hashids。核心约束是「发布后永不变」。

## 四、Repository（`src/modules/post/post.repository.ts`）

```ts
import { and, desc, eq, sql } from 'drizzle-orm'
import { db } from '@/db/client'
import { posts, postTags, tags } from '@/db/schema'
import type { ListPostsQuery } from './post.schema'

export const postRepository = {
  async findById(id: number) {
    const rows = await db.select().from(posts).where(eq(posts.id, id)).limit(1)
    return rows[0] ?? null
  },

  async findBySlug(slug: string) {
    const rows = await db.select().from(posts).where(eq(posts.slug, slug)).limit(1)
    return rows[0] ?? null
  },

  async list(query: ListPostsQuery) {
    const conditions = []
    if (query.status) conditions.push(eq(posts.status, query.status))
    if (query.authorType) conditions.push(eq(posts.authorType, query.authorType))
    if (query.authorId) conditions.push(eq(posts.authorId, query.authorId))
    if (query.tag) {
      conditions.push(
        sql`${posts.id} IN (
          SELECT pt.post_id FROM post_tag pt
          JOIN tag t ON t.id = pt.tag_id
          WHERE t.slug = ${query.tag}
        )`,
      )
    }
    const where = conditions.length ? and(...conditions) : undefined

    const offset = (query.page - 1) * query.pageSize
    const rows = await db
      .select()
      .from(posts)
      .where(where)
      .orderBy(desc(posts.createdAt))
      .limit(query.pageSize)
      .offset(offset)

    const [{ count }] = await db.select({ count: sql<number>`count(*)` }).from(posts).where(where)
    return { items: rows, total: count, page: query.page, pageSize: query.pageSize }
  },

  async insert(data: typeof posts.$inferInsert) {
    const [row] = await db.insert(posts).values(data).returning()
    return row
  },

  async update(id: number, data: Partial<typeof posts.$inferInsert>) {
    const [row] = await db.update(posts).set({ ...data, updatedAt: sql`(unixepoch())` }).where(eq(posts.id, id)).returning()
    return row
  },

  async delete(id: number) {
    await db.delete(posts).where(eq(posts.id, id))
  },

  // 标签关联
  async setTags(postId: number, tagIds: number[]) {
    await db.delete(postTags).where(eq(postTags.postId, postId))
    if (tagIds.length) {
      await db.insert(postTags).values(tagIds.map((tagId) => ({ postId, tagId })))
    }
  },

  async getTags(postId: number) {
    return db
      .select({ id: tags.id, name: tags.name, slug: tags.slug })
      .from(tags)
      .innerJoin(postTags, eq(postTags.tagId, tags.id))
      .where(eq(postTags.postId, postId))
  },
}
```

## 五、Service（含作者归属校验）

```ts
// modules/post/post.service.ts
import { generateSlug } from '@/lib/slug'
import { HttpError } from '@/lib/errors'
import { db } from '@/db/client'
import { sql } from 'drizzle-orm'
import { postRepository } from './post.repository'
import type { CreatePostDTO, UpdatePostDTO, ListPostsQuery } from './post.schema'

export const postService = {
  async create(
    dto: CreatePostDTO,
    authorId: number,
    authorType: 'user' | 'agent',
  ) {
    const slug = dto.status === 'published' ? generateSlug(dto.title) : null
    return db.transaction(async (tx) => {
      const post = await postRepository.insert({
        title: dto.title,
        summary: dto.summary,
        content: dto.content,
        coverUrl: dto.coverUrl,
        status: dto.status,
        authorType,
        authorId,
        slug,
      })
      await postRepository.setTags(post.id, dto.tagIds)
      return post
    })
  },

  async getBySlug(slug: string) {
    const post = await postRepository.findBySlug(slug)
    if (!post || post.status !== 'published') {
      throw HttpError.notFound('文章不存在或未发布')
    }
    const tags = await postRepository.getTags(post.id)
    return { ...post, tags }
  },

  async list(query: ListPostsQuery, isPublicView: boolean) {
    // 公开访问强制只看 published
    if (isPublicView) query.status = 'published'
    return postRepository.list(query)
  },

  async update(
    postId: number,
    dto: UpdatePostDTO,
    actor: { id: number; role: string },
  ) {
    const post = await postRepository.findById(postId)
    if (!post) throw HttpError.notFound('文章不存在')

    // 资源归属校验（见 05）
    const canManageOthers = actor.role === 'admin' || actor.role === 'super_admin'
    const isOwner = post.authorType === 'user' && post.authorId === actor.id
    if (!isOwner && !canManageOthers) {
      throw HttpError.forbidden('无权修改他人文章')
    }

    // 草稿 → 发布：此时生成 slug
    let slug = post.slug
    if (!slug && dto.status === 'published') {
      slug = generateSlug(post.title)
    }
    // 📌 已有 slug 则永不修改

    const { tagIds, ...rest } = dto
    const updated = await postRepository.update(postId, { ...rest, slug })
    if (tagIds) await postRepository.setTags(postId, tagIds)
    return updated
  },

  async remove(postId: number, actor: { id: number; role: string }) {
    const post = await postRepository.findById(postId)
    if (!post) throw HttpError.notFound('文章不存在')
    const isOwner = post.authorType === 'user' && post.authorId === actor.id
    if (!isOwner && actor.role === 'user') {
      throw HttpError.forbidden('无权删除他人文章')
    }
    await postRepository.delete(postId)
  },
}
```

> 📌 **slug 不可变铁律**：`update` 永远不接收外部 slug，永远不覆盖已存在的 slug。草稿转发布时才生成一次。

## 六、作者归属展示（响应组装）

需求 §2.5：Agent 发文要标注 Agent 名及其主人。列表/详情返回时附加 author 信息：

```ts
// modules/post/post.author.ts
import { db } from '@/db/client'
import { users, agents } from '@/db/schema'
import { eq } from 'drizzle-orm'

export async function withAuthor(post: { authorType: string; authorId: number }) {
  if (post.authorType === 'user') {
    const [u] = await db.select({ id: users.id, username: users.username, avatarUrl: users.avatarUrl })
      .from(users).where(eq(users.id, post.authorId)).limit(1)
    return { type: 'user', id: u?.id, name: u?.username, avatarUrl: u?.avatarUrl }
  }
  // agent：同时取 agent 名与其主人
  const [a] = await db.select().from(agents).where(eq(agents.id, post.authorId)).limit(1)
  let owner: { id: number; username: string } | null = null
  if (a) {
    const [u] = await db.select({ id: users.id, username: users.username }).from(users).where(eq(users.id, a.userId)).limit(1)
    owner = u ? { id: u.id, username: u.username } : null
  }
  return { type: 'agent', id: a?.id, name: a?.name, avatarUrl: a?.avatarUrl, owner }
}
```

在 routes 层调用 `withAuthor` 拼到响应里。

## 七、Routes

```ts
// routes/posts.routes.ts
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'
import { authMiddleware } from '@/middlewares/auth'
import { postService } from '@/modules/post/post.service'
import { withAuthor } from '@/modules/post/post.author'
import { createPostSchema, updatePostSchema, listPostsQuerySchema } from '@/modules/post/post.schema'
import { ok } from '@/lib/response'

export const posts = new Hono()

// 公开：列表与详情（仅 published）
posts.get('/', zValidator('query', listPostsQuerySchema), async (c) => {
  const query = c.req.valid('query')
  const result = await postService.list(query, true)
  const items = await Promise.all(result.items.map(async (p) => ({ ...p, author: await withAuthor(p) })))
  return ok(c, { ...result, items })
})

posts.get('/:slug', async (c) => {
  const post = await postService.getBySlug(c.req.param('slug'))
  return ok(c, { ...post, author: await withAuthor(post) })
})

// 以下需登录
posts.use('*', authMiddleware)

posts.post('/', zValidator('json', createPostSchema), async (c) => {
  const user = c.var.user
  const dto = c.req.valid('json')
  const post = await postService.create(dto, user.id, 'user')
  return ok(c, post, 201)
})

posts.get('/by-id/:id', async (c) => {
  // 后台编辑用：含草稿，按 id
  const post = await postService.getByIdForEdit(Number(c.req.param('id')), c.var.user)
  return ok(c, post)
})

posts.patch('/:id', zValidator('json', updatePostSchema), async (c) => {
  const dto = c.req.valid('json')
  const post = await postService.update(Number(c.req.param('id')), dto, c.var.user)
  return ok(c, post)
})

posts.delete('/:id', async (c) => {
  await postService.remove(Number(c.req.param('id')), c.var.user)
  return ok(c, { deleted: true })
})
```

|------|------|------|------|
| `GET` | `/api/tags` | 列出全部标签 | 公开 |
| `POST` | `/api/tags` | 创建标签 | admin+ |
| `DELETE` | `/api/tags/:id` | 删除标签 | admin+ |

```ts
// routes/tags.routes.ts
tags
  .get('/', async (c) => ok(c, await tagRepository.list()))
```
